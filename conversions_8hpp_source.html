96 inline value to_emacs(expected_type_t<uintmax_t>, envw nv, uintmax_t n) {
97  CPPEMACS_CHECK_VERSION(nv, 27);
98  using namespace detail;
99  // uintmax_t may overflow intmax_t (bad), so fallback to the bigint version
100  if (n > static_cast<uintmax_t>(std::numeric_limits<intmax_t>::max())) {
101  emacs_limb_t magnitude[uintmax_limb_count] = {0};
102  if constexpr (uintmax_limb_count == 1) {
103  magnitude[0] = static_cast<emacs_limb_t>(n);
104  } else {
105  for (int ii = 0; ii < uintmax_limb_count && n > 0;
106  ++ii, n >>= std::numeric_limits<emacs_limb_t>::digits) {
107  magnitude[ii] = n & std::numeric_limits<emacs_limb_t>::max();
108  }
109  }
110  return nv.make_big_integer(1, uintmax_limb_count, magnitude);
111  } else {
112  return to_emacs(expected_type_t<intmax_t>{}, nv, static_cast<intmax_t>(n));
113  }
114 }

118 template <typename Int, detail::enable_if_t<detail::is_integral_smaller_than_intmax<Int>::value, bool> = true>
119 inline Int from_emacs(expected_type_t<Int>, envw nv, value val) {
120  intmax_t int_val = nv.extract_integer(val);
121  nv.maybe_non_local_exit();
122  if (int_val < static_cast<intmax_t>(std::numeric_limits<Int>::min())
123  || int_val > static_cast<intmax_t>(std::numeric_limits<Int>::max())) {
124  throw std::runtime_error("Integer out of range");
125  }
126  return static_cast<Int>(int_val);
127 }

129 #ifdef CPPEMACS_HAS_UINTMAX_CONVERSION
130 inline uintmax_t from_emacs(expected_type_t<uintmax_t>, envw nv, value val) {
131  CPPEMACS_CHECK_VERSION(nv, 27);
132  using namespace detail;
133  // always use bigint conversion
134  int sign = 0;
135  ptrdiff_t count = uintmax_limb_count;
136  emacs_limb_t magnitude[uintmax_limb_count] = {0};
137  if (!nv.extract_big_integer(val, &sign, &count, magnitude)
138  || sign < 0) {
139  nv.maybe_non_local_exit();
140  throw std::runtime_error("Integer out of range");
141  }
142  if constexpr (std::is_same_v<emacs_limb_t, uintmax_t>) {
143  return magnitude[0];
144  } else {
145  constexpr size_t ndigits =
146  std::numeric_limits<uintmax_t>::digits
147  % std::numeric_limits<emacs_limb_t>::digits;
148  if (magnitude[uintmax_limb_count - 1] >= (1 << ndigits)) {
149  throw std::runtime_error("Integer out of range");
150  }
151  uintmax_t ret = 0;
152  for (int ii = uintmax_limb_count - 1; ii >= 0; --ii) {
153  ret <<= std::numeric_limits<emacs_limb_t>::digits;
154  ret |= static_cast<uintmax_t>(magnitude[ii]);
155  }
156  return ret;
157  }
158 }
159 #endif
