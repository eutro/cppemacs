72 template <typename T>
73 struct is_integral_smaller_than_intmax : std::integral_constant<bool, (
74  std::is_integral<T>::value &&
75  (std::numeric_limits<T>::digits <= std::numeric_limits<intmax_t>::digits)
76 )> {};
77 
78 #if (EMACS_MAJOR_VERSION >= 27) && !defined(CPPEMACS_DOXYGEN_RUNNING)
80 static constexpr ptrdiff_t uintmax_limb_count =
81  1 + // ceil division
82  (std::numeric_limits<emacs_limb_t>::digits - 1) /
83  std::numeric_limits<uintmax_t>::digits;
84 #define CPPEMACS_UINTMAX_ONE_LIMB (EMACS_LIMB_MAX >= UINTMAX_MAX)
85 static_assert(CPPEMACS_UINTMAX_ONE_LIMB == (detail::uintmax_limb_count == 1), "Inconsistent limb size detection");

127 inline value to_emacs(expected_type_t<uintmax_t>, envw nv, uintmax_t n) {
128  nv.check_compatible<27>();
129  using namespace detail;
130  // uintmax_t may overflow intmax_t (bad), so fallback to the bigint version
131  if (n > static_cast<uintmax_t>(std::numeric_limits<intmax_t>::max())) {
132  emacs_limb_t magnitude[uintmax_limb_count] = {0};
133 #if CPPEMACS_UINTMAX_ONE_LIMB
134  magnitude[0] = static_cast<emacs_limb_t>(n);
135 #else
136  for (int ii = 0; ii < uintmax_limb_count && n > 0;
137  ++ii, n >>= std::numeric_limits<emacs_limb_t>::digits) {
138  magnitude[ii] = n & std::numeric_limits<emacs_limb_t>::max();
139  }
140 #endif
141  return nv.make_big_integer(1, uintmax_limb_count, magnitude);
142  } else {
143  return to_emacs(expected_type_t<intmax_t>{}, nv, static_cast<intmax_t>(n));
144  }
145 }
146 #endif

167 template <typename Int, detail::enable_if_t<detail::is_integral_smaller_than_intmax<Int>::value, bool> = true>
168 inline Int from_emacs(expected_type_t<Int>, envw nv, value val) {
169  intmax_t int_val = nv.extract_integer(val);
170  nv.maybe_non_local_exit();
171  if (int_val < static_cast<intmax_t>(std::numeric_limits<Int>::min())
172  || int_val > static_cast<intmax_t>(std::numeric_limits<Int>::max())) {
173  detail::throw_out_of_range<Int>(nv, val);
174  }
175  return static_cast<Int>(int_val);
176 }

181 inline uintmax_t from_emacs(expected_type_t<uintmax_t>, envw nv, value val) {
182  nv.check_compatible<27>();
183  using namespace detail;
184  // always use bigint conversion
185  int sign = 0;
186  ptrdiff_t count = uintmax_limb_count;
187  emacs_limb_t magnitude[uintmax_limb_count] = {0};
188  if (!nv.extract_big_integer(val, &sign, &count, magnitude)
189  || sign < 0) {
190  nv.maybe_non_local_exit();
191  detail::throw_out_of_range<uintmax_t>(nv, val);
192  }
193  constexpr size_t extra_digits =
194  std::numeric_limits<uintmax_t>::digits
195  % std::numeric_limits<emacs_limb_t>::digits;
196  // uintmax_t is (by definition) the biggest integer type, and it
197  // would be bizarre for this not to be a multiple of size_t (which
198  // is what emacs_limb_t) is
199  static_assert(extra_digits == 0,
200  "UINTMAX_DIGITS must be a multiple of EMACS_LIMB_DIGITS\n"
201  "Please tell me what platform you are using!");
202 #if CPPEMACS_UINTMAX_ONE_LIMB
203  return static_cast<uintmax_t>(magnitude[0]);
204 #else
205  uintmax_t ret = 0;
206  for (int ii = uintmax_limb_count - 1; ii >= 0; --ii) {
207  ret <<= std::numeric_limits<emacs_limb_t>::digits;
208  ret |= static_cast<uintmax_t>(magnitude[ii]);
209  }
210  return ret;
211 #endif
212 }
213 #endif

248 inline value to_emacs(expected_type_t<mpz_class>, envw nv, const mpz_class &zc) {
249  nv.check_compatible<27>();
250  nv.maybe_non_local_exit();
251 
252  mpz_srcptr z = zc.get_mpz_t();
253  if (mpz_fits_slong_p(z)) {
254  return nv.make_integer(mpz_get_si(z));
255  }
256 
257  int sign = mpz_sgn(z);
258  size_t count = 0;
259  size_t nwords = mpz_sizeinbase(z, 2) / std::numeric_limits<emacs_limb_t>::digits;
260  std::unique_ptr<emacs_limb_t[]> magnitude(new emacs_limb_t[nwords]);
261  mpz_export(
262  magnitude.get(), &count,
263  -1, sizeof(emacs_limb_t), 0, 0, // LE limbs, native byte order, full words
264  z
265  );
266 
267  return nv.make_big_integer(sign, count, magnitude.get());
268 }

276 inline mpz_class from_emacs(expected_type_t<mpz_class>, envw nv, value x) {
277  nv.check_compatible<27>();
278 
279  int sign = 0;
280  ptrdiff_t count = 0;
281  if (nv.extract_big_integer(x, &sign, &count, nullptr)) {
282  if (sign == 0) {
283  return 0;
284  }
285 
286  std::unique_ptr<emacs_limb_t[]> magnitude(new emacs_limb_t[count]);
287  if (nv.extract_big_integer(x, &sign, &count, magnitude.get())) {
288  mpz_class ret;
289 
290  mpz_ptr z = ret.get_mpz_t();
291  mpz_import(
292  z, count,
293  -1, sizeof(emacs_limb_t), 0, 0, // LE limbs, native byte order, full words
294  magnitude.get()
295  );
296 
297  if (sign == -1) {
298  mpz_neg(z, z);
299  }
300 
301  return ret;
302  }
303  }
304 
305  nv.maybe_non_local_exit();
306  throw std::runtime_error("Bigint conversion failed");
307 }
