54  struct table_key {
55  /* must have static lifetime */
56  const char *name; size_t len;
57  table_key(const char *name, size_t len): name(name), len(len) {}
58  bool operator==(const table_key &o) const {
59  return len == o.len &&
60  (name == o.name || std::strncmp(name, o.name, len) == 0);
61  }
62  };
63  struct hasher {
64  size_t operator()(const table_key &key) const noexcept {
65 #ifdef __cpp_lib_string_view
66  return std::hash<std::string_view>()(std::string_view(key.name, key.len));
67 #else
68  // Java-like hashCode
69  size_t hash = 7;
70  for (size_t i = 0; i < key.len; ++i) {
71  hash = (31 * hash) + static_cast<size_t>(key.name[i]);
72  }
73  return hash;
74 #endif
75  }
76  };
77  std::unordered_map<table_key, value, hasher> table;
78 
79 public:
80  value intern(const char *name, size_t len) noexcept {
81  auto pair = table.emplace(
82  std::piecewise_construct,
83  std::forward_as_tuple(name, len),
84  std::forward_as_tuple(nullptr));
85  if (pair.second) {
86  return pair.first->second = envw::intern(name);
87  } else {
88  return pair.first->second;
89  }

174  }

182  emacs_finalizer fin = nv.get_user_finalizer(val);

186  }
187  return user_ptr<T, Deleter>(reinterpret_cast<T*>(nv.get_user_ptr(val)));

202 template <typename F, typename = void>
203 struct module_function_repr {
204  static F &extract(void *ptr) noexcept { return *reinterpret_cast<F *>(ptr); }
205  static value make(
206  envw nv, ptrdiff_t min_arity, ptrdiff_t max_arity,
207  emacs_function fun, const char *doc, F &&f
208  ) {
209  nv.maybe_non_local_exit();
210 
211  // The strategy here is to bind the module function to a new
212  // symbol, which then has a finalizer attached. We own the
213  // instance of F precisely until the finalizer is created,
214  // at which point it is owned by the GC.
215 
216  std::unique_ptr<F> fptr(new F(std::move(f)));
217  void *fptr_data = reinterpret_cast<void *>(fptr.get());
218  emacs_finalizer fin = [](void *data) noexcept { delete reinterpret_cast<F *>(data); };
219 
220  value retfn = nv.make_function(
221  min_arity, max_arity, fun, doc,
222  fptr_data); // fptr still owned by us for a bit
223 
224 #if (EMACS_MAJOR_VERSION >= 28) // we have module finalizers, maybe
225  if (nv->size >= sizeof(emacs_env_28)) {
226  // we have function finalizers, just attach it
227  nv.set_function_finalizer(retfn, fin);
228  nv.maybe_non_local_exit();
229  fptr.release(); // now managed by the GC
230  return retfn;
231  }
232 #endif
233  value gensym = nv.intern("gensym");
234  // bind the module function to a new symbol, for which we can add a finalizer
235  value func_sym = (nv->*gensym)(std::string("c++fun-"));
236  (nv->*nv.intern("defalias"))(func_sym, retfn);
237 
238  // make a user pointer as a finalizer
239  value finalizer = nv.make_user_ptr(fin, fptr_data);
240  nv.maybe_non_local_exit();
241  fptr.release(); // the finalizer was created successfully, F is now managed by the GC
242 
243  // attach the finalizer, so that the lifetimes of `retfn` and the finalizer are tied
244  // (unless users do something stupid!)
245  nv.funcall(nv.intern("put"), {func_sym, (nv->*gensym)(), finalizer});
246 
247  return func_sym; // an alias to our module function in a fresh uninterned symbol
248  }
249 };

353 template <ptrdiff_t MinArity, ptrdiff_t MaxArity, bool IsVariadic, typename F>
354 struct spread_invoker {
355  F f;
356  template <typename Arg>
357  spread_invoker(Arg &&arg): f(std::forward<Arg>(arg)) {}
358 
359 private:
360  template <bool IsProvided> detail::enable_if_t<IsProvided, cell>
361  arg_or_default(envw nv, value *args, size_t idx) noexcept(false) { return nv->*args[idx]; }
362  template <bool IsProvided> detail::enable_if_t<!IsProvided, unprovided_value_t>
363  arg_or_default(envw, value *, size_t) noexcept { return {}; }
364 
365  template <size_t NArgs, size_t...Idx, bool IsVar = IsVariadic> detail::enable_if_t<!IsVar, value>
366  invoke_with_arity(detail::index_sequence<Idx...>, envw nv, value *args) noexcept(false)
367  { return nv->*f(nv, arg_or_default<Idx < NArgs>(nv, args, Idx)...); }
368  template <size_t...Idx, bool IsVar = IsVariadic> detail::enable_if_t<!IsVar, value>
369  invoke_variadic(detail::index_sequence<Idx...>, envw, ptrdiff_t, value *) noexcept(false) { return {}; }
370 
371  template <size_t NArgs, size_t...Idx, bool IsVar = IsVariadic> detail::enable_if_t<IsVar, value>
372  invoke_with_arity(detail::index_sequence<Idx...>, envw nv, value *args) noexcept(false)
373  { return nv->*f(nv, arg_or_default<Idx < NArgs>(nv, args, Idx)..., {nullptr, 0}); }
374  template <size_t...Idx, bool IsVar = IsVariadic> detail::enable_if_t<IsVar, value>
375  invoke_variadic(detail::index_sequence<Idx...>, envw nv, ptrdiff_t nargs, value *args) noexcept(false)
376  { return nv->*f(nv, arg_or_default<true>(nv, args, Idx)..., {args + MaxArity, nargs - MaxArity}); }
377 
378  template <size_t...OffIdx>
379  value invoke(
380  detail::index_sequence<OffIdx...>,
381  envw nv, ptrdiff_t nargs, value *args
382  ) noexcept(false) {
383  using arity_seq = detail::make_index_sequence<MaxArity>;
384  if (IsVariadic && nargs > MaxArity) {
385  return invoke_variadic(arity_seq(), nv, nargs, args);
386  } else if (nargs < MinArity || nargs > MaxArity) {
387  // should have been caught by Emacs
388  throw std::runtime_error("Bad arity");
389  } else {
390  static constexpr value (*Vt[])(spread_invoker *, envw, value*) = {
391  ([](spread_invoker *self, envw nv, value *args)
392  { return self->invoke_with_arity<MinArity + OffIdx>(arity_seq(), nv, args); })
393  ...
394  };
395  return Vt[nargs](this, nv, args);
396  }
397  }
398 
399 public:
400  value operator()(envw nv, ptrdiff_t nargs, value *args) noexcept {
401  return nv.run_catching([&]() noexcept(false) {
402  return invoke(
403  detail::make_index_sequence<MaxArity - MinArity + 1>(),
404  nv, nargs, args);
405  });
406  }
407 };
408 }
