167 template <typename F, typename = void>
168 struct module_function_repr {
170  static F &extract(void *ptr) noexcept { return *reinterpret_cast<F *>(ptr); }
172  static value make(
173  envw nv, ptrdiff_t min_arity, ptrdiff_t max_arity,
174  value (*fun)(emacs_env*, ptrdiff_t, value*, void*) noexcept,
175  const char *doc, F &&f
176  ) noexcept(std::is_nothrow_move_constructible<F>::value) {
177  if (nv.non_local_exit_check()) return nullptr;
178 
179  // The strategy here is to bind the module function to a new
180  // symbol, which then has a finalizer attached. We own the
181  // instance of F precisely until the finalizer is created,
182  // at which point it is owned by the GC.
183 
184  static_assert(std::is_move_constructible<F>::value, "Type must be move constructible");
185  std::unique_ptr<F> fptr(new F(std::move(f)));
186  void *fptr_data = reinterpret_cast<void *>(fptr.get());
187  void (*fin)(void*) noexcept = user_ptr<F>::fin;
188 
189  value retfn = nv.make_function(
190  min_arity, max_arity, fun, doc,
191  fptr_data); // fptr still owned by us for a bit
192 
193 #if (EMACS_MAJOR_VERSION >= 28) // we have module finalizers, maybe
194  if (nv.is_compatible<28>()) {
195  // we have function finalizers, just attach it
196  nv.set_function_finalizer(retfn, fin);
197  if (nv.non_local_exit_check()) return nullptr;
198  fptr.release(); // now managed by the GC
199  return retfn;
200  }
201 #endif
202  value make_symbol = nv.intern("make-symbol");
203  // bind the module function to a new symbol, for which we can add a finalizer
204  value func_sym = (nv->*make_symbol)(nv.make_string("cpp--finalized-fun"));
205  (nv->*nv.intern("defalias"))(func_sym, retfn);
206 
207  // make a user pointer as a finalizer
208  value finalizer = nv.make_user_ptr(fin, fptr_data);
209  if (nv.non_local_exit_check()) return nullptr;
210  fptr.release(); // the finalizer was created successfully, F is now managed by the GC
211 
212  // attach the finalizer, so that the lifetimes of `retfn` and the finalizer are tied
213  // (unless users do something stupid!)
214  nv.funcall(nv.intern("put"), {func_sym, (nv->*"cpp--data-ptr"), finalizer});
215  if (nv.non_local_exit_check()) return nullptr; // make sure that we don't return if the lifetimes weren't tied
216 
217  return func_sym; // an alias to our module function in a fresh uninterned symbol
218  }
219 };

478 template <ptrdiff_t MinArity, ptrdiff_t MaxArity, bool IsVariadic, typename F>
479 struct spread_invoker {
481  F f;
483  template <typename Arg>
484  spread_invoker(Arg &&arg): f(std::forward<Arg>(arg)) {}
485 
486 private:
487  using arity_seq = detail::make_index_sequence<MaxArity>;
488  using caller = spreader_caller<IsVariadic, spread_invoker>;
489  friend caller;
490 
491  template <typename ...Args>
492  value asserted_call(envw nv, Args&&...args)
493  {
494  auto ret = f(nv, std::forward<Args>(args)...);
495  return nv->*ret;
496  }
497 
498  template <size_t...OffIdx>
499  value invoke(
500  detail::index_sequence<OffIdx...>,
501  envw nv, ptrdiff_t nargs, value *args
502  ) noexcept(false) {
503  if (IsVariadic && nargs > MaxArity) {
504  return caller::template invoke_variadic<MaxArity>(arity_seq(), *this, nv, nargs, args);
505  } else if (nargs < MinArity || nargs > MaxArity) {
506  // should have been caught by Emacs
507  throw std::runtime_error("Bad arity");
508  } else {
509  static constexpr value (*Vt[])(spread_invoker *, envw, value*) = {
510  ([](spread_invoker *self, envw nv, value *args)
511  { return caller::template invoke_with_arity<MinArity + OffIdx>(arity_seq(), *self, nv, args); })
512  ...
513  };
514  return Vt[nargs](this, nv, args);
515  }
516  }
517 
518 public:
520  value operator()(envw nv, ptrdiff_t nargs, value *args) noexcept {
521  return nv.run_catching([&]() -> value {
522  return invoke(
523  detail::make_index_sequence<MaxArity - MinArity + 1>(),
524  nv, nargs, args);
525  });
526  }
527 };
528 }
