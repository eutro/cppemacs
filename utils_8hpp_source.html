43template <typename T, typename Deleter = std::default_delete<T>>
44struct user_ptr {
45 static_assert(!std::is_reference<T>::value, "Must not be a reference type");
46 static_assert(std::is_default_constructible<Deleter>::value, "Deleter must be default-constructible");
47private:
48 T *ptr;
49
50public:
52 user_ptr(T *ptr) noexcept: ptr(ptr) {}
53
55 T *get() const noexcept { return ptr; }
57 T *operator->() const noexcept { return ptr; }
59 T &operator*() const noexcept { return *ptr; }
60
68 static void fin(void *ptr) noexcept {
69 try {
70 Deleter()(reinterpret_cast<T*>(ptr));
71 } catch (...) {
72 // TODO: log warnings?
73 }
74 }
75
88 friend value to_emacs(expected_type_t<user_ptr>, envw nv, const user_ptr &ptr) noexcept {
89 value ret = nullptr;
90 void *raw = reinterpret_cast<void*>(ptr.ptr);
91 if (nv.non_local_exit_check()
92 || (ret = nv.make_user_ptr(user_ptr<T, Deleter>::fin, raw),
93 nv.non_local_exit_check())) {
94 fin(raw);
95 }
96 return ret;
97 }
98
108 friend user_ptr from_emacs(expected_type_t<user_ptr>, envw nv, value val) {
109 emacs_finalizer fin = nv.get_user_finalizer(val);
110 nv.maybe_non_local_exit();
111 if (fin != user_ptr<T, Deleter>::fin) {
112 throw std::runtime_error("User ptr type mismatch");
113 }
114 return user_ptr<T, Deleter>(reinterpret_cast<T*>(nv.get_user_ptr(val)));
115 }
116};

130make_user_ptr(Args &&...args) { return user_ptr<T>(new T(std::forward<Args>(args)...)); }

143template <typename F, typename = void>
144struct module_function_repr {
146 static F &extract(void *ptr) noexcept { return *reinterpret_cast<F *>(ptr); }
148 static value make(
149 envw nv, ptrdiff_t min_arity, ptrdiff_t max_arity,
150 value (*fun)(emacs_env*, ptrdiff_t, value*, void*) noexcept,
151 const char *doc, F &&f
152 ) noexcept(std::is_nothrow_move_constructible<F>::value) {
153 if (nv.non_local_exit_check()) return nullptr;
154
155 // The strategy here is to bind the module function to a new
156 // symbol, which then has a finalizer attached. We own the
157 // instance of F precisely until the finalizer is created,
158 // at which point it is owned by the GC.
159
160 static_assert(std::is_move_constructible<F>::value, "Type must be move constructible");
161 std::unique_ptr<F> fptr(new F(std::move(f)));
162 void *fptr_data = reinterpret_cast<void *>(fptr.get());
163 void (*fin)(void*) noexcept = user_ptr<F>::fin;
164
165 value retfn = nv.make_function(
166 min_arity, max_arity, fun, doc,
167 fptr_data); // fptr still owned by us for a bit
168
169#if (EMACS_MAJOR_VERSION >= 28) // we have module finalizers, maybe
170 if (nv.is_compatible<28>()) {
171 // we have function finalizers, just attach it
172 nv.set_function_finalizer(retfn, fin);
173 if (nv.non_local_exit_check()) return nullptr;
174 fptr.release(); // now managed by the GC
175 return retfn;
176 }
177#endif
178 value make_symbol = nv.intern("make-symbol");
179 // bind the module function to a new symbol, for which we can add a finalizer
180 value func_sym = (nv->*make_symbol)(nv.make_string("cpp--finalized-fun"));
181 (nv->*nv.intern("defalias"))(func_sym, retfn);
182
183 // make a user pointer as a finalizer
184 value finalizer = nv.make_user_ptr(fin, fptr_data);
185 if (nv.non_local_exit_check()) return nullptr;
186 fptr.release(); // the finalizer was created successfully, F is now managed by the GC
187
188 // attach the finalizer, so that the lifetimes of `retfn` and the finalizer are tied
189 // (unless users do something stupid!)
190 nv.funcall(nv.intern("put"), {func_sym, (nv->*"cpp--data-ptr"), finalizer});
191 if (nv.non_local_exit_check()) return nullptr; // make sure that we don't return if the lifetimes weren't tied
192
193 return func_sym; // an alias to our module function in a fresh uninterned symbol
194 }
195};

253struct module_function {
254 static_assert(!std::is_reference<F>::value, "Must not be a reference type");
255#ifdef CPPEMACS_HAVE_IS_INVOCABLE
256 static_assert(std::is_invocable_r_v<value, F&, emacs_env *, ptrdiff_t, value *>,
257 "F must be invocable with Emacs function arguments");
258#endif

274 using data_repr = module_function_repr<F>;
275
277 ptrdiff_t min_arity;
279 ptrdiff_t max_arity;
282 const char *doc;
284 F f;

292 template <typename...Args>
293 module_function(
294 ptrdiff_t min_arity, ptrdiff_t max_arity,
295 const char *doc,
296 Args &&...args):
297 min_arity(min_arity),
298 max_arity(max_arity),
299 doc(doc),
300 f(std::forward<Args>(args)...)
301 {}
302
305 static value invoke(emacs_env *nv, ptrdiff_t nargs, value *args, void *data) noexcept {
306 return envw(nv).run_catching(
307 [&]() noexcept(
308 noexcept(value(data_repr::extract(data)(nv, nargs, args)))
309 ) -> value {
310 return data_repr::extract(data)(nv, nargs, args);
311 });
312 }
313
319 friend value to_emacs(expected_type_t<module_function>, envw nv, module_function &&func) {
320 return data_repr::make(
321 nv, func.min_arity, func.max_arity, &invoke, func.doc,
322 std::move(func.f));
323 }
324};

454template <ptrdiff_t MinArity, ptrdiff_t MaxArity, bool IsVariadic, typename F>
455struct spread_invoker {
457 F f;
459 template <typename Arg>
460 spread_invoker(Arg &&arg): f(std::forward<Arg>(arg)) {}
461
462private:
463 using arity_seq = detail::make_index_sequence<MaxArity>;
464 using caller = spreader_caller<IsVariadic, spread_invoker>;
465 friend caller;
466
467 template <typename ...Args>
468 value asserted_call(envw nv, Args&&...args)
469 {
470 auto ret = f(nv, std::forward<Args>(args)...);
471 return nv->*ret;
472 }
473
474 template <size_t Arity>
475 static value vt_fun(spread_invoker *self, envw nv, value *args)
476 { return caller::template invoke_with_arity<Arity>(arity_seq(), *self, nv, args); }
477 using vt_funtype = value (*)(spread_invoker *, envw, value *);
478
479 template <size_t...OffIdx>
480 value invoke(
481 detail::index_sequence<OffIdx...>,
482 envw nv, ptrdiff_t nargs, value *args
483 ) noexcept(false) {
484 if (IsVariadic && nargs > MaxArity) {
485 return caller::template invoke_variadic<MaxArity>(arity_seq(), *this, nv, nargs, args);
486 } else if (nargs < MinArity || nargs > MaxArity) {
487 // should have been caught by Emacs
488 throw std::runtime_error("Bad arity");
489 } else {
490 static constexpr vt_funtype Vt[] = { &spread_invoker::vt_fun<MinArity + OffIdx> ... };
491 return Vt[nargs - MinArity](this, nv, args);
492 }
493 }
494
495public:
497 value operator()(envw nv, ptrdiff_t nargs, value *args) noexcept {
498 return nv.run_catching([&]() -> value {
499 return invoke(
500 detail::make_index_sequence<MaxArity - MinArity + 1>(),
501 nv, nargs, args);
502 });
503 }
504};
505}

560template <ptrdiff_t MinArity, ptrdiff_t MaxArity, bool IsVariadic, typename F>
561auto make_spreader_function(
562 spreader_arity<MinArity, MaxArity, IsVariadic> ar,
563 const char *doc, F &&f
564)
565#if !CPPEMACS_HAVE_RETURN_TYPE_DEDUCTION
566 -> module_function<detail::spread_invoker<MinArity, MaxArity, IsVariadic, detail::remove_reference_t<F>>>
567#endif
568{
569 constexpr bool is_variadic = decltype(ar)::is_variadic;
570 static_assert(is_variadic || MaxArity >= MinArity, "MaxArity must be greater than or equal to MinArity");
571 static_assert(MinArity >= 0, "MinArity must be nonnegative");
572 return make_module_function(
573 MinArity, MaxArity, doc,
574 detail::spread_invoker<MinArity, MaxArity, is_variadic,
575 detail::remove_reference_t<F>>(std::forward<F>(f)));
576}
