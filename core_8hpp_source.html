1066 template <bool Unbox = CPPEMACS_ENABLE_EXCEPTION_BOXING>
1067 void rethrow_non_local_exit() const noexcept(false) {
1068 value symbol, data;
1069 funcall_exit kind = non_local_exit_get(symbol, data);
1070 if (!kind) return;
1071 non_local_exit_clear();
1072 if (kind == funcall_exit::signal_) {
1073 CPPEMACS_MAYBE_IF_CONSTEXPR (Unbox) {
1074 if (eq(symbol, intern("error"))) {
1075 value msg = funcall(intern("car"), {data});
1076 std::string msg_string = from_emacs(expected_type_t<std::string>{}, *this, msg);
1077 if (non_local_exit_check()) {
1078 non_local_exit_clear();
1079 } else {
1080 throw std::runtime_error(std::move(msg_string));
1081 }
1082 } if (eq(symbol, intern("cppemacs--exception"))) {
1083 auto eptr = reinterpret_cast<std::exception_ptr *>(get_user_ptr(data));
1084 if (non_local_exit_check()) {
1085 non_local_exit_clear();
1086 } else if (eptr && *eptr) {
1087 std::rethrow_exception(*eptr);
1088 }
1089 }
1090 }
1091 throw signalled(symbol, data);
1092 } else {
1093 throw thrown(symbol, data);
1094 }
1095 }

1398 /* Extracted to a non-generic (over F) function, to reduce code size, and
1399 * for the (not correctness-bearing) `static bool is_init;`. */
1400 template <bool Box>
1401 void run_catching_handle_current(std::integral_constant<bool, Box>) const noexcept {
1402 if (non_local_exit_check()) return;
1403 try {
1404 try {
1405 throw;
1406 } catch (const signalled &s) {
1407 non_local_exit_signal(s.symbol, s.data);
1408 } catch (const thrown &s) {
1409 non_local_exit_throw(s.symbol, s.data);
1410 } catch (const non_local_exit &) {
1411 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1412 throw;
1413 } else {
1414 funcall(
1415 intern("error"),
1416 {make_string("Expected non-local exit")}
1417 );
1418 }
1419 } catch (const std::exception &err) {
1420 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1421 // std::runtime_error is thrown directly as error
1422 if (typeid(err) != typeid(std::runtime_error)) {
1423 throw;
1424 }
