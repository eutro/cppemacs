944  template <bool Unbox = CPPEMACS_ENABLE_EXCEPTION_BOXING>
945  void rethrow_non_local_exit() const noexcept(false) {
946  value symbol, data;
947  funcall_exit kind = non_local_exit_get(symbol, data);
948  if (!kind) return;
949  non_local_exit_clear();
950  if (kind == funcall_exit::signal_) {
951  CPPEMACS_MAYBE_IF_CONSTEXPR (Unbox) {
952  if (eq(symbol, intern("error"))) {
953  value msg = funcall(intern("car"), {data});
954  std::string msg_string = from_emacs(expected_type_t<std::string>{}, *this, msg);
955  if (non_local_exit_check()) {
956  non_local_exit_clear();
957  } else {
958  throw std::runtime_error(std::move(msg_string));
959  }
960  } if (eq(symbol, intern("cppemacs--exception"))) {
961  auto eptr = reinterpret_cast<std::exception_ptr *>(get_user_ptr(data));
962  if (non_local_exit_check()) {
963  non_local_exit_clear();
964  } else if (eptr && *eptr) {
965  std::rethrow_exception(*eptr);
966  }
967  }
968  }
969  throw signal(symbol, data);
970  } else {
971  throw thrown(symbol, data);
972  }
973  }

1264  /* Extracted to a non-generic (over F) function, to reduce code size, and
1265  * for the (not correctness-bearing) `static bool is_init;`. */
1266  template <bool Box>
1267  void run_catching_handle_current(std::integral_constant<bool, Box>) const noexcept {
1268  if (non_local_exit_check()) return;
1269  try {
1270  throw;
1271  } catch (const signal &s) {
1272  non_local_exit_signal(s.symbol, s.data);
1273  } catch (const thrown &s) {
1274  non_local_exit_throw(s.symbol, s.data);
1275  } catch (const non_local_exit &) {
1276  CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1277  throw;
1278  } else {
1279  funcall(
1280  intern("error"),
1281  {make_string("Expected non-local exit")}
1282  );
1283  }
1284  } catch (const std::exception &err) {
1285  CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1286  // std::runtime_error is thrown directly as error
1287  if (typeid(err) != typeid(std::runtime_error)) {
1288  throw;
1289  }
1290  }
1291  const char *str = err.what();
1292  funcall(
1293  intern("error"),
1294  {make_string(str, std::char_traits<char>::length(str))}
1295  );
1296  } catch (...) {
1297  CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1298  void (*fin)(void *) noexcept = [](void *v) noexcept
1299  { delete reinterpret_cast<std::exception_ptr*>(v); };
1300  auto uptr_ptr = new std::exception_ptr(std::current_exception());
1301 
1302  static bool is_init = false;
1303  value tag = intern("cppemacs--exception");
1304  if (!is_init) {
1305  funcall(intern("define-error"), {
1306  tag,
1307  make_string("Opaque C++ exception")
1308  });
1309  is_init = true;
1310  }
1311 
1312  value uptr = make_user_ptr(fin, uptr_ptr);
1313  funcall(intern("signal"), {tag, uptr});
1314  } else {
1315  funcall(
1316  intern("error"),
1317  {make_string("Unrecognised exception")}
1318  );
1319  }
1320  }
1321  assert(non_local_exit_check());
1322  }
1323 
1324  // F might throw
1325  template <typename BoxV, typename F, detail::enable_if_t<!noexcept(std::declval<F>()()), bool> = true>
1326  auto run_catching_internal(BoxV, F &&f) const noexcept -> detail::invoke_result_t<F> {
1327  using return_type = detail::invoke_result_t<F>;
1328  static_assert(
1329  std::is_void<return_type>::value ||
1330  std::is_nothrow_default_constructible<return_type>::value,
1331  "Return type must be void or nothrow default constructible");
1332 
1333  try {
1334  return std::forward<F>(f)();
1335  } catch(...) {
1336  run_catching_handle_current(BoxV{});
1337  }
1338  return return_type();
1339  }
