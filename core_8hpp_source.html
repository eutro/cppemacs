990 template <bool Unbox = CPPEMACS_ENABLE_EXCEPTION_BOXING>
991 void rethrow_non_local_exit() const noexcept(false) {
992 value symbol, data;
993 funcall_exit kind = non_local_exit_get(symbol, data);
994 if (!kind) return;
995 non_local_exit_clear();
996 if (kind == funcall_exit::signal_) {
997 CPPEMACS_MAYBE_IF_CONSTEXPR (Unbox) {
998 if (eq(symbol, intern("error"))) {
999 value msg = funcall(intern("car"), {data});
1000 std::string msg_string = from_emacs(expected_type_t<std::string>{}, *this, msg);
1001 if (non_local_exit_check()) {
1002 non_local_exit_clear();
1003 } else {
1004 throw std::runtime_error(std::move(msg_string));
1005 }
1006 } if (eq(symbol, intern("cppemacs--exception"))) {
1007 auto eptr = reinterpret_cast<std::exception_ptr *>(get_user_ptr(data));
1008 if (non_local_exit_check()) {
1009 non_local_exit_clear();
1010 } else if (eptr && *eptr) {
1011 std::rethrow_exception(*eptr);
1012 }
1013 }
1014 }
1015 throw signal(symbol, data);
1016 } else {
1017 throw thrown(symbol, data);
1018 }
1019 }

1310 /* Extracted to a non-generic (over F) function, to reduce code size, and
1311 * for the (not correctness-bearing) `static bool is_init;`. */
1312 template <bool Box>
1313 void run_catching_handle_current(std::integral_constant<bool, Box>) const noexcept {
1314 if (non_local_exit_check()) return;
1315 try {
1316 throw;
1317 } catch (const signal &s) {
1318 non_local_exit_signal(s.symbol, s.data);
1319 } catch (const thrown &s) {
1320 non_local_exit_throw(s.symbol, s.data);
1321 } catch (const non_local_exit &) {
1322 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1323 throw;
1324 } else {
1325 funcall(
1326 intern("error"),
1327 {make_string("Expected non-local exit")}
1328 );
1329 }
1330 } catch (const std::exception &err) {
1331 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1332 // std::runtime_error is thrown directly as error
1333 if (typeid(err) != typeid(std::runtime_error)) {
1334 throw;
1335 }
1336 }
1337 const char *str = err.what();
1338 funcall(
1339 intern("error"),
1340 {make_string(str, std::char_traits<char>::length(str))}
1341 );
1342 } catch (...) {
1343 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1344 void (*fin)(void *) noexcept = [](void *v) noexcept
1345 { delete reinterpret_cast<std::exception_ptr*>(v); };
1346 auto uptr_ptr = new std::exception_ptr(std::current_exception());
1347
1348 static bool is_init = false;
1349 value tag = intern("cppemacs--exception");
1350 if (!is_init) {
1351 funcall(intern("define-error"), {
1352 tag,
1353 make_string("Opaque C++ exception")
1354 });
1355 is_init = true;
1356 }
1357
1358 value uptr = make_user_ptr(fin, uptr_ptr);
1359 funcall(intern("signal"), {tag, uptr});
1360 } else {
1361 funcall(
1362 intern("error"),
1363 {make_string("Unrecognised exception")}
1364 );
1365 }
1366 }
1367 assert(non_local_exit_check());
1368 }

1371 template <typename BoxV, typename F, detail::enable_if_t<!noexcept(std::declval<F>()()), bool> = true>
1372 auto run_catching_internal(BoxV, F &&f) const noexcept -> detail::invoke_result_t<F> {
1373 using return_type = detail::invoke_result_t<F>;
1374 static_assert(
1375 std::is_void<return_type>::value ||
1376 std::is_nothrow_default_constructible<return_type>::value,
1377 "Return type must be void or nothrow default constructible");
1378
1379 try {
1380 return std::forward<F>(f)();
1381 } catch(...) {
1382 run_catching_handle_current(BoxV{});
1383 }
1384 return return_type();
1385 }

1472 FPtrType fptr;
1473 std::exception_ptr exn;
1474 };

1476 value func = make_function(1, 1, [](emacs_env *raw_env, ptrdiff_t, value *args, void *data) noexcept {
1477 envw env = raw_env;
1478 fun_data &fdata = *reinterpret_cast<fun_data*>(data);
1479 try {
1480 std::forward<F>(*fdata.fptr)(env);
1481 } catch (...) {
1482 fdata.exn = std::current_exception();
1483 }
1484 return args[0];
1485 }, nullptr, reinterpret_cast<void *>(&fdata));
1486
1487 funcall(func, 1, &func);
1488 if (!is_noexcept) {
1489 if (fdata.exn) {
1490 std::rethrow_exception(std::move(fdata.exn));
1491 }
1492
1493 maybe_non_local_exit();
1494 }
1495 }
