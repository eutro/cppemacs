1470 template <class Boxer, typename F, detail::enable_if_t<noexcept(std::declval<F>()()), bool> = true>
1471 auto run_catching_internal(Boxer &&, F &&f) const noexcept -> detail::invoke_result_t<F> {
1472 return std::forward<F>(f)();
1473 }
1474
1475 // F might throw
1476 template <class Boxer, typename F, detail::enable_if_t<!noexcept(std::declval<F>()()), bool> = true>
1477 auto run_catching_internal(Boxer &&b, F &&f) const noexcept -> detail::invoke_result_t<F> {
1478 using return_type = detail::invoke_result_t<F>;
1479 static_assert(
1480 std::is_void<return_type>::value ||
1481 std::is_nothrow_default_constructible<return_type>::value,
1482 "Return type must be void or nothrow default constructible");
1483
1484 try {

1617template <bool Box> inline void do_box_exceptions(emacs_env *raw) noexcept {
1618 envw env = raw;
1619 if (env.non_local_exit_check()) return;
1620 try {
1621 try {
1622 throw;
1623 } catch (const signalled &s) {
1624 env.non_local_exit_signal(s.symbol, s.data);
1625 } catch (const thrown &s) {
1626 env.non_local_exit_throw(s.symbol, s.data);
1627 } catch (const non_local_exit &) {
1628 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1629 throw;
1630 } else {
1631 env.funcall(
1632 env.intern("error"),
1633 {env.make_string("Expected non-local exit")}
1634 );
1635 }
1636 } catch (const std::exception &err) {
1637 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1638 // std::runtime_error is thrown directly as error
1639 if (typeid(err) != typeid(std::runtime_error)) {
1640 throw;
1641 }
1642 }
1643 const char *str = err.what();
1644 env.funcall(
1645 env.intern("error"),
1646 {env.make_string(str, std::char_traits<char>::length(str))}
1647 );
1648 }
1649 } catch (...) {
1650 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1651 static bool is_init = false;
1652 value tag = env.intern("cppemacs--exception");
1653 if (!is_init) {
1654 env.funcall(env.intern("define-error"), {
1655 tag,
1656 env.make_string("Opaque C++ exception")
1657 });
1658 is_init = true;
1659 }
1660
1661 if (!env.non_local_exit_check()) {
1662 auto eptr = new std::exception_ptr(std::current_exception());
1663 value uptr = env.make_user_ptr(
1664 &exception_ptr_fin,
1665 static_cast<void *>(eptr)
1666 );
1667 if (env.non_local_exit_check()) {
1668 // delete if it didn't make it to the GC
1669 delete eptr;
1670 } else {
1671 env.funcall(env.intern("signal"), {tag, uptr});
1672 }
1673 }
1674 } else {
1675 env.funcall(
1676 env.intern("error"),
1677 {env.make_string("Unrecognised exception")}
1678 );
1679 }
1680 }

1683template <bool Box> inline void do_rethrow(emacs_env *raw) {
1684 envw env = raw;
1685 value symbol, data;
1686 funcall_exit kind = env.non_local_exit_get(symbol, data);
1687 if (!kind) return;
1688 env.non_local_exit_clear();
1689 if (kind == funcall_exit::signal_) {
1690 CPPEMACS_MAYBE_IF_CONSTEXPR (Box) {
1691 if (env.eq(symbol, env.intern("error"))) {
1692 value msg = env.funcall(env.intern("car"), {data});
1693 std::string msg_string = from_emacs(expected_type_t<std::string>{}, env, msg);
1694 if (env.non_local_exit_check()) {
1695 env.non_local_exit_clear();
1696 } else {
1697 throw std::runtime_error(std::move(msg_string));
1698 }
1699 } if (env.eq(symbol, env.intern("cppemacs--exception"))) {
1700 auto eptr = reinterpret_cast<std::exception_ptr *>(env.get_user_ptr(data));
1701 if (env.non_local_exit_check()) {
1702 env.non_local_exit_clear();
1703 } else if (eptr && *eptr) {
1704 std::rethrow_exception(*eptr);
1705 }
1706 }
1707 }
1708 throw signalled(symbol, data);
1709 } else {
1710 throw thrown(symbol, data);
1711 }
1712}
