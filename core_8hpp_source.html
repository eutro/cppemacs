1072 template <bool Unbox = CPPEMACS_ENABLE_EXCEPTION_BOXING>
1073 void rethrow_non_local_exit() const noexcept(false) {
1074 value symbol, data;
1075 funcall_exit kind = non_local_exit_get(symbol, data);
1076 if (!kind) return;
1077 non_local_exit_clear();
1078 if (kind == funcall_exit::signal_) {
1079 CPPEMACS_MAYBE_IF_CONSTEXPR (Unbox) {
1080 if (eq(symbol, intern("error"))) {
1081 value msg = funcall(intern("car"), {data});
1082 std::string msg_string = from_emacs(expected_type_t<std::string>{}, *this, msg);
1083 if (non_local_exit_check()) {
1084 non_local_exit_clear();
1085 } else {
1086 throw std::runtime_error(std::move(msg_string));
1087 }
1088 } if (eq(symbol, intern("cppemacs--exception"))) {
1089 auto eptr = reinterpret_cast<std::exception_ptr *>(get_user_ptr(data));
1090 if (non_local_exit_check()) {
1091 non_local_exit_clear();
1092 } else if (eptr && *eptr) {
1093 std::rethrow_exception(*eptr);
1094 }
1095 }
1096 }
1097 throw signalled(symbol, data);
1098 } else {
1099 throw thrown(symbol, data);
1100 }
1101 }

1404 /* Extracted to a non-generic (over F) function, to reduce code size, and
1405 * for the (not correctness-bearing) `static bool is_init;`. */
1406 template <bool Box>
1407 void run_catching_handle_current(std::integral_constant<bool, Box>) const noexcept {
1408 if (non_local_exit_check()) return;
1409 try {
1410 try {
1411 throw;
1412 } catch (const signalled &s) {
1413 non_local_exit_signal(s.symbol, s.data);
1414 } catch (const thrown &s) {
1415 non_local_exit_throw(s.symbol, s.data);
1416 } catch (const non_local_exit &) {
1417 CPPEMACS_MAYBE_IF_CONSTEXPR(Box) {
1418 throw;
1419 } else {
1420 funcall(
1421 intern("error"),
1422 {make_string("Expected non-local exit")}
1423 );
1424 }
