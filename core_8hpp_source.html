294  template <typename F>
295  auto run_catching(F f) const noexcept -> detail::invoke_result_t<F> {
296  using return_type = detail::invoke_result_t<F>;
297  static_assert(
298  std::is_void<return_type>::value ||
299  std::is_nothrow_default_constructible<return_type>::value,
300  "Return type must be void or nothrow default constructible");
301  try {
302  return f();
303  } catch (const signal &s) {
304  non_local_exit_clear();
305  non_local_exit_signal(s.symbol, s.data);
306  } catch (const thrown &s) {
307  non_local_exit_clear();
308  non_local_exit_throw(s.symbol, s.data);
309  } catch (const non_local_exit &) {
310  if (!non_local_exit_check()) {
311  funcall(
312  intern("error"),
313  {make_string("Expected non-local exit")}
314  );
315  }
316  } catch (std::runtime_error &err) {
317  non_local_exit_clear();
318  const char *str = err.what();
319  funcall(
320  intern("error"),
321  {make_string(str, std::strlen(str))}
322  );
323  } catch (...) {
324  non_local_exit_clear();
325  funcall(
326  intern("error"),
327  {make_string("Unrecognised exception")}
328  );
329  }
330  assert(non_local_exit_check());
331  return return_type();
332  }
